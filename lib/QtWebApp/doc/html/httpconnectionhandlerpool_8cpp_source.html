55  if (pool.count()<maxConnectionHandlers)

57  freeHandler=new HttpConnectionHandler(settings,requestHandler,sslConfiguration);

68 {

70  int idleCounter=0;
71  mutex.lock();
72  foreach(HttpConnectionHandler* handler, pool)

83  }
84  }
